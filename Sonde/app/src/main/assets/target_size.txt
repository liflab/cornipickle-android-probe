
"""
  @name Accommodate floating action button
  @description The button should not be hidden
  @severity Warning
"""
For each $x in $(Button) (
    For each $y in $(Button) (

    (((($x's top addto $x's bottom) / 2) addto ($x's height / 2)) addto ((($y's top addto $y's bottom) / 2) addto ($y's height / 2))) is less than 20
    )
).
